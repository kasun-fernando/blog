Numerical integration with python

We discuss how to implement two basic ODE solving algorithms (Euler, Runge-Kutta) using python.

Euler Method

  • Given the first order ODE: $$\frac{dx}{dt}=f(x,t),\,\,\, x(0)=x_0$$ one can write its solution in the integral form $$x(t)=x_0+\int_0^tf(x(s),s)\, ds.$$
  • Discretize the $t-$axis into intervals of length $\Delta t$, write $t_0=0$, $t_j - t_{j-1}=\Delta t$ for $j \geq 1$, and take the approximate value of $x$ at $t_j$ to be $x_j$.
  • This leads to the first order approximation of the solution $$x_j = x_{j-1} + f(x_{j-1},t_{j-1}) \Delta t,\,\,\, j \geq 1.$$

  • Global error is $\mathcal{O}(\Delta t)$ while the local error is $\mathcal{O}((\Delta t)^2)$

  • Example 1: $$\frac{dx}{dt}=-x,\,\,\, x(0)=1 \implies x(t)=e^{-t},\,\,\, t \geq 0.$$

import numpy as np
import matplotlib.pyplot as plt

dt, tmin, tmax = 0.001, 0.0, 10.0 # initialize parameters
N = int((tmax - tmin)/dt) # number of steps

t = np.linspace(tmin,tmax,N) # divide t-axis to N steps within  the range
x = np.zeros(N) # initialize the array of approx. x values

xreal = np.exp(-t) # real solution
plt.plot(t, xreal, label="Exact", lw=1) # plot real solution linewidth=1

x[0]=1.0 # initial condition

for i in range(N-1):
    x[i+1]=x[i]-x[i]*dt  # Euler method algo
    
plt.plot(t, x, ls='--', lw=1, label='Numerical') # plotting numerical solution
# plt.plot(t,x/xreal, label='Ratio')
plt.legend()
plt.show()

# plt.plot(t,x/xreal, label='Ratio')

# plt.legend()
# plt.show()

plt.plot(t,100*(np.abs(x-xreal))/xreal, label='Relative error')

plt.legend()
plt.show()

Harmonic Oscillator

A particle attached to a string (with its the other end fixed to the origin) in 2-D can be modelled using two variables, its postition $R(t)$ and its velocity $V(t)$. The systems of ODEs that govern the motion is $$\frac{dR}{dt}=V$$ $$m\frac{dV}{dt}=-\zeta V -\kappa R$$ where $\kappa$ is the spring constant and $\zeta$ is the fircition constant (air friction). The total energy of the system is given by $$E = \frac{1}{2}mV^2+\frac{1}{2} \kappa R^2.$$

  • Euler Method: In general, the Euler method is not ideal to find numerical solutions for Harmonic Oscillators, but let's use it first. $$R_{i}=R_{i-1}+V_i \Delta t$$ $$V_{i}=\left(1-\frac{\zeta}{m}\right)V_{i-1} \Delta t-\kappa R_{i-1}\Delta t$$ Note that the the approximate total energy given by the scheme $$E_i=\frac{1}{2}mV^2_i+\frac{1}{2}\kappa R_i^2$$ does not remain constant in the case $\zeta=0$ (undpamped oscialltor) even though in theory it should. 
%matplotlib nbagg
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.animation as animation 
plt.style.use('ggplot')

# imported required libraries

dim = 2 
StepNo = 1001 # number of steps
R = np.zeros(dim) # position 
V = np.zeros(dim) # velocity
Rall = np.zeros([dim,StepNo]) # position at each time-stamp
Vall = np.zeros([dim,StepNo]) # velocity at each time-stamp
E = np.zeros(StepNo) # total energy at each time-stamp
time = np.zeros(StepNo) # time stamps

#defined all the variables

def init(): 
    particles.set_data([], [])
    line.set_data([], [])
    title.set_text(r'')
    return particles,line,title

#a function to initialize animation 

def animate(i): 
    global R, V, F, Rall, Vall, E, time
    R = R + V*dt
    V = V*(1-(zeta/m)*dt) - (k/m)*dt*R  # Euler scheme
    Rall[0:dim,i]=R
    Vall[0:dim,i]=V
    time[i]=i*dt
    E[i]=0.5*m*np.linalg.norm(V)**2 + 0.5*k*np.linalg.norm(R)**2
    particles.set_data(R[0],R[1])
    line.set_data(Rall[0,0:i], Rall[1,0:i])
    title.set_text(r"$t={0:.2f}, E_t={1:.3f}$".format(i*dt,E[i]))
    return particles,line,title

#a function to animate Euler algo's solution

m, k, zeta = 1.0, 1.0, 0.0 # parameters in the system

R[0], R[1] = 1., 1. # R_x(0) and R_y(0)
V[0], V[1] = 0., 1. # V_x(0) and V_y(0)

dt = 0.1*np.sqrt(k/m) 
box = 4 

fig, ax = plt.subplots(figsize=(5,5))
ax = plt.axes(xlim=(-box/2,box/2), ylim=(-box/2,box/2))
particles, = ax.plot([], [], 'ko', ms=10)
line, = ax.plot([], [], lw=1)
title = ax.text(0.5, 1.05, r'', transform = ax.transAxes, va = 'center') 
anim = animation.FuncAnimation(fig, animate, init_func=init, frames=StepNo, interval=5, blit=True, repeat=False)

# anim.save('movie.mp4',fps=20,dpi=400)

fig, ax = plt.subplots(figsize=(7,3))
ax.set_xlabel(r"$t$", fontsize=14)
ax.plot(time,Rall[0]) # plot R_x(t)
ax.plot(time,Rall[1]) # plot R_y(t)
ax.plot(time,E) # plot E, constant if zeta=0)
ax.legend([r'$R_x(t)$',r'$R_y(t)$',r'$E(t)$'], fontsize=14)
plt.show()

  • Runge-Kutta Method

    • We get more precise approximate solutions by higher order methods. In this case, we use a 4th order method. $$ y_{i+1} = y_i+\int_{{t_i}}^{t_{i+1}} f(y(s),s) \, ds\,,$$ yields the following approximation. $$ y_{i+1}\simeq y_i+\frac{1}{6}\Delta t [k_1 +2k_2+2k_3+k_4] $$ where the intermediate variables are defined as follows: $$ k_1 = f(t_i,y_i)\,, $$ $$ k_2 = f\left(t_n + \frac{\Delta t}{2}, y_i+\frac{\Delta t}{2} k_1 \right)\,, $$ $$ k_3 = f\left(t_n + \frac{\Delta t}{2}, y_i+\frac{\Delta t}{2}k_2 \right)\,,$$ $$ k_4 = f(t_i+\Delta t, y_i+\Delta t k_3)\,.$$
    • We illustrate how the scheme is implemented.
    • The idea is to use temporary storage variables Ri and Vi for $i=1,2,3,4$ (corresponding to $k_i$ in the scheme).
    • The function animate should be appropriately changed.
    • In this case, the approximate energy remains constant (up to at least four decimal places).
    • This is because the global error is $\mathcal{O}((\Delta t)^4)$ while the local error is $\mathcal{O}((\Delta t)^5)$.
%matplotlib nbagg
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.animation as animation 
plt.style.use('ggplot')

# imported required libraries

dim = 2 
StepNo = 1001 # number of steps
R = np.zeros(dim) # position 
V = np.zeros(dim) # velocity
Rall = np.zeros([dim,StepNo]) # position at each time-stamp
Vall = np.zeros([dim,StepNo]) # velocity at each time-stamp
E = np.zeros(StepNo) # total energy at each time-stamp
time = np.zeros(StepNo) # time stamps

#defined all the variables

def init(): 
    particles.set_data([], [])
    line.set_data([], [])
    title.set_text(r'')
    return particles,line,title

R1 = np.zeros(dim) # intermediate variables
R2 = np.zeros(dim)
R3 = np.zeros(dim)
R4 = np.zeros(dim)
V1 = np.zeros(dim)
V2 = np.zeros(dim)
V3 = np.zeros(dim)
V4 = np.zeros(dim)

def animate(i): 
    global R, V, F, Rall, Vall, E, time
    
    R1=V
    V1=-(zeta/m)*V-(k/m)*R
    R2=V+dt*V1/2
    V2=-(zeta/m)*(V+V1*dt/2)-(k/m)*(R+R1*dt/2)
    R3=V+dt*V2/2
    V3=-(zeta/m)*(V+V2*dt/2)-(k/m)*(R+R2*dt/2)
    R4=V+dt*V3
    V4=-(zeta/m)*(V+V3*dt)-(k/m)*(R+R3*dt)
    
    V = V + (V1+2*V2+2*V3+V4)*dt/6
    R = R + (R1+2*R2+2*R3+R4)*dt/6
    
    Rall[0:dim,i]=R
    Vall[0:dim,i]=V
    time[i]=i*dt
    E[i]=0.5*m*np.linalg.norm(V)**2 + 0.5*k*np.linalg.norm(R)**2
    particles.set_data(R[0],R[1])
    line.set_data(Rall[0,0:i], Rall[1,0:i])
    title.set_text(r"$t={0:.2f}, E_t={1:.4f}$".format(i*dt,E[i]))
    return particles,line,title

m, k, zeta = 1.0, 1.0, 0.0 # parameters in the system

R[0], R[1] = 1., 1. # R_x(0) and R_y(0)
V[0], V[1] = 0., 1. # V_x(0) and V_y(0)

dt = 0.1*np.sqrt(k/m) 
box = 4 

fig, ax = plt.subplots(figsize=(5,5))
ax = plt.axes(xlim=(-box/2,box/2), ylim=(-box/2,box/2))
particles, = ax.plot([], [], 'ko', ms=10)
line, = ax.plot([], [], lw=1)
title = ax.text(0.5, 1.05, r'', transform = ax.transAxes, va = 'center') 
anim = animation.FuncAnimation(fig, animate, init_func=init, frames=StepNo, interval=5, blit=True, repeat=False)

fig, ax = plt.subplots(figsize=(7,3))
ax.set_xlabel(r"$t$", fontsize=14)
ax.plot(time,Rall[0]) # plot R_x(t)
ax.plot(time,Rall[1]) # plot R_y(t)
ax.plot(time,E) # plot E, constant if zeta=0)
ax.legend([r'$R_x(t)$',r'$R_y(t)$',r'$E(t)$'], fontsize=14)
plt.show()
